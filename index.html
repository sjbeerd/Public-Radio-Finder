<script>
// Start automatically on load
window.onload = function() {
if (navigator.geolocation) {
navigator.geolocation.getCurrentPosition(findStation, handleError);
} else {
document.getElementById('status').innerText = "Geolocation not supported";
}
};
async function findStation(position) {
const lat = position.coords.latitude;
const lon = position.coords.longitude;
try {
const response = await fetch('https://de1.api.radio-
browser.info/json/stations/search?countrycode=US&tag=npr&hidebroken=true&limit=200');
const stations = await response.json();
let closest = null;
let minDistance = Infinity;
stations.forEach(s => {
if (s.geo_lat && s.geo_long) {
const d = getDistance(lat, lon, s.geo_lat, s.geo_long);
if (d < minDistance) {
minDistance = d;
closest = s;
}
}
});
if (closest) {
document.getElementById('status').innerText = "Closest NPR Station:";
document.getElementById('station-name').innerText = closest.name.split('-
')[0].trim();
document.getElementById('station-freq').innerText = closest.state || "Public Radio";
document.getElementById('meta').innerText = `${minDistance.toFixed(1)} miles
away`;
}
} catch (err) {
document.getElementById('status').innerText = "Error fetching data.";
}
}
function handleError() {
document.getElementById('status').innerText = "Please enable Location Access.";
}
function getDistance(lat1, lon1, lat2, lon2) {
const R = 3958.8; // Miles
const dLat = (lat2 - lat1) * Math.PI / 180;
const dLon = (lon2 - lon1) * Math.PI / 180;
const a = Math.sin(dLat/2) * Math.sin(dLat/2) +
Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) *
Math.sin(dLon/2) * Math.sin(dLon/2);
return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
}
